server {
    listen 80;
    listen [::]:80;
    listen 443 ssl;
    listen [::]:443 ssl;

    server_name chriscafe.net www.chriscafe.net;
    server_tokens off;

    include /etc/letsencrypt/options-ssl-nginx.conf;

    if ($scheme != "https") {
        return 301 https://$host$request_uri;
    }

    gzip on;
    gzip_types      text/plain application/xml;
    gzip_proxied    no-cache no-store private expired;
    gzip_min_length 1000;
    gunzip on;

    location / {
        proxy_pass         http://localhost:5001;
        proxy_http_version 1.1;
        proxy_set_header   Upgrade $http_upgrade;
        proxy_set_header   Connection keep-alive;
        proxy_set_header   Host $host;
        proxy_cache_bypass $http_upgrade;
        proxy_set_header   X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header   X-Forwarded-Proto $scheme;
        proxy_set_header      X-Real-IP $remote_addr;
        client_max_body_size    10m;
        client_body_buffer_size 128k;
        proxy_connect_timeout   90;
        proxy_send_timeout      90;
        proxy_read_timeout      90;
        proxy_buffers           32 4k;

    }

    ssl_certificate /etc/letsencrypt/live/chriscafe.net/fullchain.pem; # managed by Certbot
    ssl_certificate_key /etc/letsencrypt/live/chriscafe.net/privkey.pem; # managed by Certbot

}
